const SHEET_ID = "1qjKvOzzgSeEORH3HPwiPKXZ6b2r3m0Kxo9SKnbntZ9g"; const SHEET_NAME = "PMS Submissions"; const DRIVE_FOLDER_ID = "1n3KiTQLLSRznePc3w_fSsuN1VEls1mya"; function doGet(e) { try { const data = JSON.parse(decodeURIComponent(e.parameter.data)); // File upload request if (data.action === 'uploadFile') { return handleFileUpload(data); } // Normal form submission writeToSheet(data); return ContentService .createTextOutput(JSON.stringify({ status: "success" })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: "error", message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function handleFileUpload(data) { try { const folder = DriveApp.getFolderById(data.folderId); const blob = Utilities.newBlob( Utilities.base64Decode(data.fileData), data.mimeType, data.empName + '_' + data.quarter + '_' + data.year + '_' + data.fileName ); const file = folder.createFile(blob); return ContentService .createTextOutput(JSON.stringify({ status: "success", fileId: file.getId(), fileName: file.getName() })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: "error", message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function writeToSheet(data) { const ss = SpreadsheetApp.openById(SHEET_ID); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); addHeaders(sheet, data); } if (sheet.getLastRow() === 0) { addHeaders(sheet, data); } const row = buildRow(data); sheet.appendRow(row); formatLastRow(sheet, data); } function buildRow(data) { const row = [ data.timestamp, data.supervisor, data.empCode, data.empName, data.role, data.group, data.station, data.quarter, data.year, data.totalScore, data.maxScore, data.percentScore, ]; data.criteria.forEach(c => row.push(c.score || 0)); // Notes column — all criterion notes combined const notesSummary = data.criteria .filter(c => c.note && c.note.trim()) .map(c => c.name + ': ' + c.note) .join(' | '); row.push(notesSummary); // Attachment column row.push(data.hasAttachment ? data.attachmentName : ''); return row; } function addHeaders(sheet, data) { const headers = [ "Timestamp", "Supervisor", "Emp Code", "Employee Name", "Role", "Group", "Station", "Quarter", "Year", "Total Score", "Max Score", "% Score", ]; data.criteria.forEach(c => headers.push(c.name)); headers.push("Notes"); headers.push("Attachment"); sheet.appendRow(headers); const headerRange = sheet.getRange(1, 1, 1, headers.length); headerRange.setBackground("#1F4E79"); headerRange.setFontColor("#FFFFFF"); headerRange.setFontWeight("bold"); headerRange.setFontSize(10); headerRange.setWrap(true); sheet.setFrozenRows(1); sheet.setColumnWidth(1, 160); sheet.setColumnWidth(2, 140); sheet.setColumnWidth(3, 80); sheet.setColumnWidth(4, 160); sheet.setColumnWidth(5, 180); sheet.setColumnWidth(6, 180); sheet.setColumnWidth(7, 80); sheet.setColumnWidth(8, 70); sheet.setColumnWidth(9, 60); sheet.setColumnWidth(10, 90); sheet.setColumnWidth(11, 80); sheet.setColumnWidth(12, 80); for (let i = 13; i <= 12 + data.criteria.length; i++) { sheet.setColumnWidth(i, 70); } sheet.setColumnWidth(13 + data.criteria.length, 400); sheet.setColumnWidth(14 + data.criteria.length, 200); } function formatLastRow(sheet, data) { const lastRow = sheet.getLastRow(); const totalCols = 14 + data.criteria.length; const range = sheet.getRange(lastRow, 1, 1, totalCols); range.setBackground(lastRow % 2 === 0 ? "#F5F7FA" : "#FFFFFF"); // Colour % score cell (col 12) const pctCell = sheet.getRange(lastRow, 12); const pct = pctCell.getValue(); pctCell.setNumberFormat("0\"%\""); if (pct >= 85) pctCell.setBackground("#DEEAF1").setFontColor("#1F4E79").setFontWeight("bold"); else if (pct >= 70) pctCell.setBackground("#E2EFDA").setFontColor("#375623").setFontWeight("bold"); else if (pct >= 50) pctCell.setBackground("#FFF2CC").setFontColor("#7F6000").setFontWeight("bold"); else pctCell.setBackground("#FCE4D6").setFontColor("#C00000").setFontWeight("bold"); range.setFontFamily("Arial"); range.setFontSize(10); range.setVerticalAlignment("middle"); range.setBorder(true, true, true, true, true, true, "#E0E0E0", SpreadsheetApp.BorderStyle.SOLID); }
